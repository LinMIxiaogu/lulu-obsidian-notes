 131.分割回文串

### 题目说明

给你一个字符串 `s`，请你将 `s` 分割成一些子串，使每个子串都是回文串。

返回 `s` 所有可能的分割方案。

示例：

```text
输入: s = "aab"
输出: [["a","a","b"],["aa","b"]]
```

### Java 解答

```java
import java.util.ArrayList;
import java.util.List;

class Solution {
    List<List<String>> res = new ArrayList<>();
    boolean[][] dp;

    public List<List<String>> partition(String s) {
        int n = s.length();
        dp = new boolean[n][n];

        for (int i = n - 1; i >= 0; i--) {
            for (int j = i; j < n; j++) {
                if (i == j || (s.charAt(i) == s.charAt(j) && (j - i == 1 || dp[i + 1][j - 1]))) {
                    dp[i][j] = true;
                }
            }
        }

        backTrack(s, 0, n, new ArrayList<>());
        return res;
    }

    private void backTrack(String s, int index, int n, List<String> list) {
        if (index == n) {
            res.add(new ArrayList<>(list));
            return;
        }
        for (int j = index; j < n; j++) {
            if (dp[index][j]) {
                list.add(s.substring(index, j + 1));
                backTrack(s, j + 1, n, list);
                list.remove(list.size() - 1);
            }
        }
    }
}
```
